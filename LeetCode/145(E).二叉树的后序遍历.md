## 思路
- 后序遍历顺序是：左-右-根
- 遍历方式主要有递归和队列遍历


## 递归
```JavaScript
var postorderTraversal = function (root) {
  let res = []

  function postorder(node) {
    if (!node) {
      return
    }

    postorder(node.left)
    postorder(node.right)
    res.push(node.val)
  }

  postorder(root)
  return res
};
```

## 栈
```JavaScript
var postorderTraversal = function (root) {
  if (!root) {
    return []
  }
  let res = []
  let stack = []

  while (root || stack.length) {
    res.push(root.val)

    if (root.left) {
      stack.push(root.left)
    }
    
    if (root.right) {
      stack.push(root.right)
    }

    root = stack.pop()
  }
  
  res.reverse()
  return res
}
```
