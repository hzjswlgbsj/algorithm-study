## 方法一：每次遍历调换位置
```JavaScript
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function (head) {
  let prev = null
  let curNode = head
  while (curNode) {
    const next = curNode.next
    curNode.next = prev
    prev = curNode
    curNode = next
  }

  return prev
};
```
